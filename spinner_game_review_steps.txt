4. Add Subtle Metal Scratches

function drawScratch(c, x, y, r, a1, a2) {
    let g = c.createConicGradient(a1, x, y);

    g.addColorStop(0.00, "rgba(255,255,255,0)");
    g.addColorStop(0.02, "rgba(255,255,255,0.5)");
    g.addColorStop(0.05, "rgba(255,255,255,0.15)");
    g.addColorStop(0.10, "rgba(255,255,255,0)");

    c.strokeStyle = g;
    c.lineWidth = 1;

    c.beginPath();
    c.arc(x, y, r, a1, a2);
    c.stroke();
}
